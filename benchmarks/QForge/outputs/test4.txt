# Import necessary libraries
from fastapi import FastAPI, Depends, HTTPException, status
from fastapi.security import OAuth2PasswordBearer
from pydantic import BaseModel
import jwt

app = FastAPI()

# Define the data model for the user
class User(BaseModel):
    username: str
    password: str

# Define the data model for the token
class Token(BaseModel):
    access_token: str
    token_type: str

# Define the data model for the token data
class TokenData(BaseModel):
    username: str | None = None

# Secret key for JWT encoding
SECRET_KEY = "your_secret_key"

# Algorithm for JWT encoding
ALGORITHM = "HS256"

# Password to authenticate the user
PASSWORD = "your_password"

# Create an instance of OAuth2PasswordBearer
oauth2_scheme = OAuth2PasswordBearer(tokenUrl="token")

# Function to create a token
def create_access_token(data: dict):
    encoded_jwt = jwt.encode(data, SECRET_KEY, algorithm=ALGORITHM)
    return encoded_jwt

# Function to authenticate the user
def authenticate_user(username: str, password: str):
    if username != "admin" or password != PASSWORD:
        return False
    return True

# Route to generate a token
@app.post("/token")
async def login_for_access_token(form_data: OAuth2PasswordRequestForm = Depends()):
    if not authenticate_user(form_data.username, form_data.password):
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Incorrect username or password",
            headers={"WWW-Authenticate": "Bearer"},
        )
    access_token_expires = timedelta(minutes=30)
    access_token = create_access_token(
        data={"sub": form_data.username},
    )
    return {"access_token": access_token, "token_type": "bearer"}

# Dependency for accessing user data with authentication
def get_current_user(token: str = Depends(oauth2_scheme)):
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Could not validate credentials",
        headers={"WWW-Authenticate": "Bearer"},
    )
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        username: str = payload.get("sub")
        if username is None:
            raise credentials_exception
        token_data = TokenData(username=username)
    except JWTError:
        raise credentials_exception
    user = get_user(fake_users_db, token_data.username)
    if user is None:
        raise credentials_exception
    return user
